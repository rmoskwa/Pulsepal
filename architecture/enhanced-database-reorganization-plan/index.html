    -- 1. knowledge_base: Core content with rich metadata
CREATE TABLE knowledge_base (
    id BIGSERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    source_url TEXT,

    -- Content classification
    content_type VARCHAR(50) CHECK (content_type IN (
        'sequence_implementation',  -- Complete sequence examples
        'pulseq_function',         -- Core Pulseq functions (mr.*)
        'vendor_adaptation',       -- GE/Siemens specific
        'educational',            -- Tutorials, workshops
        'mri_utility',           -- MRI-specific helpers
        'concept_explanation'    -- Theory, physics
    )),

    -- Quality and relevance scoring
    quality_score INTEGER DEFAULT 50,  -- 0-100 scale
    pulseq_version VARCHAR(20),       -- '1.4.0', '2.0.0', etc.
    language VARCHAR(10),              -- 'matlab', 'python'

    -- Rich metadata for better search
    metadata JSONB DEFAULT '{}',
    /*
    Example metadata structure:
    {
        "sequence_type": "EPI",           -- GRE, EPI, MPRAGE, etc.
        "difficulty": "intermediate",     -- beginner, intermediate, advanced
        "has_reconstruction": true,
        "vendor_specific": ["GE", "Siemens"],
        "pulseq_functions": ["mr.makeSincPulse", "seq.addBlock"],
        "file_path": "original/path/to/file.m",
        "github_stars": 245,              -- Repository popularity
        "last_updated": "2024-06-15",
        "dependencies": ["Signal Processing Toolbox"],
        "tested_scanners": ["GE Premier", "Siemens Prisma"]
    }
    */

    -- Vector embedding for semantic search
    embedding vector(768),

    -- Timestamps
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),

    -- Indexes
    INDEX idx_content_type (content_type),
    INDEX idx_quality_score (quality_score),
    INDEX idx_language (language),
    INDEX idx_embedding_vector (embedding) USING ivfflat
);

-- 2. validated_patterns: Curated, verified code patterns
CREATE TABLE validated_patterns (
    id BIGSERIAL PRIMARY KEY,

    -- Validation metadata
    verified_by VARCHAR(100),      -- Who validated this
    verification_date DATE,
    works_on_scanners TEXT[],      -- Array of tested scanners

    -- Usage tracking
    usage_count INTEGER DEFAULT 0,
    last_used TIMESTAMPTZ,

    metadata JSONB DEFAULT '{}'
);

    # Python script for intelligent content migration
import json
from typing import Dict, List
import re

def analyze_content(content: str, metadata: Dict) -> Dict:
    """Analyze content and generate rich metadata."""

    enriched_metadata = {
        "original_metadata": metadata,
        "pulseq_functions": extract_pulseq_functions(content),
        "sequence_type": detect_sequence_type(content),
        "has_vendor_code": detect_vendor_specific(content),
        "complexity_score": calculate_complexity(content),
        "has_reconstruction": "recon" in content.lower(),
        "line_count": len(content.split('\n')),
        "has_comments": detect_documentation_quality(content)
    }

    # Determine content type
    if "seq.addBlock" in content and "seq.write" in content:
        content_type = "sequence_implementation"
    elif "function" in content and "mr." in content:
        content_type = "pulseq_function"
    elif "TRID" in content or "seq2ceq" in content:
        content_type = "vendor_adaptation"
    elif "tutorial" in content.lower() or "example" in content.lower():
        content_type = "educational"
    else:
        content_type = "mri_utility"

    # Calculate quality score
    quality_score = calculate_quality_score(content, enriched_metadata)

    return {
        "content_type": content_type,
        "metadata": enriched_metadata,
        "quality_score": quality_score
    }

def extract_pulseq_functions(content: str) -> List[str]:
    """Extract all Pulseq function calls."""
    patterns = [
        r'mr\.\w+',
        r'seq\.\w+',
        r'opts\.\w+',
    ]
    functions = []
    for pattern in patterns:
        functions.extend(re.findall(pattern, content))
    return list(set(functions))

def detect_sequence_type(content: str) -> str:
    """Detect the MRI sequence type."""
    sequence_markers = {
        'EPI': ['epi', 'echo.planar'],
        'GRE': ['gradient.echo', 'gre', 'flash'],
        'TSE': ['turbo.spin', 'tse', 'fse'],
        'MPRAGE': ['mprage', 'mp.rage'],
        'DWI': ['diffusion', 'dwi', 'dti'],
        'CEST': ['cest', 'chemical.exchange'],
    }

    content_lower = content.lower()
    for seq_type, markers in sequence_markers.items():
        if any(marker in content_lower for marker in markers):
            return seq_type
    return "unknown"

def calculate_quality_score(content: str, metadata: Dict) -> int:
    """Calculate content quality score (0-100)."""
    score = 50  # Base score

    # Positive factors
    if metadata.get("has_comments"):
        score += 10
    if len(metadata.get("pulseq_functions", [])) > 5:
        score += 15
    if metadata.get("sequence_type") != "unknown":
        score += 10
    if "seq.checkTiming" in content:
        score += 10  # Proper validation
    if metadata.get("has_vendor_code"):
        score += 5  # Real-world applicability

    # Negative factors
    if metadata.get("line_count", 0) < 50:
        score -= 10  # Too simple
    if "TODO" in content or "FIXME" in content:
        score -= 15  # Incomplete
    if "deprecated" in content.lower():
        score -= 20

    return max(0, min(100, score))

    Vendor Conversion Tools: Why They're Critical¶

    The Pulseq Execution Pipeline¶

    1. User writes Pulseq code in MATLAB/Python
   ↓
2. Generate .seq file (vendor-neutral format)
   ↓
3. VENDOR CONVERSION (Critical step!)
   ↓
4. Scanner executes vendor-specific format

    What Vendor Conversion Tools Do¶

    These tools translate the universal Pulseq .seq format into vendor-specific languages that MRI scanners can execute.

    Vendor Conversion Tool Output Format Key Adaptations
    GE seq2ge, seq2ceq, PulCeq .pge, .tar TRID labels, 4μs gradient raster, segment boundaries
    Siemens Direct interpreter .seq (direct) Minimal conversion needed
    Bruker Bruker interpreter Proprietary Vendor-specific timing
    Philips Philips interpreter Proprietary Vendor-specific format

    Example: GE Conversion Process¶

    % Step 1: User writes Pulseq sequence
seq = mr.Sequence(sys);
seq.addBlock(rf, gz);
seq.addBlock(gx, adc);
seq.write('mysequence.seq');  % Universal format

% Step 2: Convert for GE scanner (REQUIRED!)
ceq = seq2ceq('mysequence.seq');  % Conversion tool
ceq.trid = [1, 1, 2, 2];  % Add GE-specific segment markers
writeceq(ceq, 'mysequence.pge');  % GE can execute this

% Without conversion, the scanner cannot run the sequence!

    Why Keep These in the Knowledge Base¶

    1. Deployment Critical: Users can't run sequences without these converters
    2. Vendor-Specific Knowledge: Each scanner has unique requirements:
    3. GE: TRID labels, timing constraints
    4. Siemens: Label extensions, reconstruction markers
    5. Hardware limitations differ by vendor
    6. Real-World Usage: The gap between "sequence works in simulation" and "sequence runs on scanner"
